      if(tag === "popular") return "Popüler";

      if(tag === "limited") return "Sınırlı";
      return "Ürün";
    }

    // ===== State =====
    let cartCount = getCartCount();
    let activeChip = "all";
    let maxPrice = 2499;
    let query = "";
    let sortMode = "popularity";

    // ===== Rendering =====
    function computeActiveFiltersText(){
      const parts = [];
      if(activeChip && activeChip !== "all") parts.push(`Etiket: ${activeChip}`);
      if(maxPrice < 2499) parts.push(`Max: ${moneyTRY(maxPrice)}`);
      if(query) parts.push(`Arama: "${query}"`);
      return parts.length ? parts.join(" • ") : "Filtre: Yok";
    }

    function applySort(list){
      const arr = [...list];
      switch(sortMode){
        case "new":        return arr.sort((a,b)=> a.created - b.created); // smaller = newer
        case "bestseller": return arr.sort((a,b)=> b.sold - a.sold);
        case "priceAsc":   return arr.sort((a,b)=> a.price - b.price);
        case "priceDesc":  return arr.sort((a,b)=> b.price - a.price);
        case "popularity":
        default:           return arr.sort((a,b)=> b.popularity - a.popularity);
      }
    }

    function filterProducts(){
      let list = PRODUCTS;

      // chip filter
      if(activeChip && activeChip !== "all"){
        list = list.filter(p => p.tag === activeChip);
      }

      // price filter
      list = list.filter(p => p.price <= maxPrice);

      // search
      if(query){
        const q = query.toLowerCase();
        list = list.filter(p => (p.name + " " + p.desc).toLowerCase().includes(q));
      }

      // sort
      list = applySort(list);
      return list;
    }

    function render(){
      $("#cartCount").textContent = cartCount;
      $("#minPriceLabel").textContent = moneyTRY(999).replace(",00","");
      $("#maxPriceLabel").textContent = moneyTRY(maxPrice).replace(",00","");
      $("#activeFilters").textContent = computeActiveFiltersText();

      const list = filterProducts();
      $("#countText").textContent = `${list.length} ürün`;

      const grid = $("#grid");
      grid.innerHTML = "";

      if(list.length === 0){
        const empty = document.createElement("div");
        empty.className = "empty";
        empty.innerHTML = `
          Sonuç bulunamadı.<br>
          Filtreleri gevşet veya aramayı değiştir.
        `;
        grid.appendChild(empty);
        return;
      }

      list.forEach(p => {
        const card = document.createElement("article");
        card.className = "p-card";
        card.innerHTML = `
          <div class="p-img" aria-hidden="true">
            <span class="p-badge ${badgeClass(p.tag)}">${badgeText(p.tag)}</span>
          </div>
          <div class="p-body">
            <p class="p-title">${p.name}</p>
            <p class="p-desc">${p.desc}</p>
            <div class="p-meta">
              <span>Hızlı Kargo</span>
              <span>${p.ship}</span>
            </div>
          </div>
          <div class="p-row">
            <div class="price">${moneyTRY(p.price)}</div>
            <div class="p-actions">
              <button class="p-btn outline" data-view="${p.id}">Ürünü İncele</button>
              <button class="p-btn" data-add="${p.id}">Sepete Ekle</button>
            </div>
          </div>
        `;
        grid.appendChild(card);
      });
    }

    // ===== Events =====
    function setActiveChip(chip){
      activeChip = chip;
      document.querySelectorAll("#chipbar .chip").forEach(el => {
        el.classList.toggle("active", el.getAttribute("data-chip") === activeChip);
      });
      render();
    }

    function resetAll(){
      activeChip = "all";
      maxPrice = 2499;
      query = "";
      sortMode = "popularity";
      $("#sortSelect").value = "popularity";
      $("#priceRange").value = "2499";
      $("#searchInput").value = "";
      setActiveChip("all");
      render();
      toast("Sıfırlandı", "Filtreler temizlendi.");
    }

    function wire(){
      // init
      $("#sortSelect").addEventListener("change", (e) => {
        sortMode = e.target.value;
        render();
      });

      $("#priceRange").addEventListener("input", (e) => {
        maxPrice = Number(e.target.value);
        render();
      });

      $("#searchInput").addEventListener("input", (e) => {
        query = e.target.value.trim();
        render();
      });

      $("#chipbar").addEventListener("click", (e) => {
        const chip = e.target.closest("[data-chip]");
        if(!chip) return;
        setActiveChip(chip.getAttribute("data-chip"));
      });

      $("#resetBtn").addEventListener("click", resetAll);

      $("#quickSortPrice").addEventListener("click", () => {
        sortMode = (sortMode === "priceAsc") ? "priceDesc" : "priceAsc";
        $("#sortSelect").value = sortMode;
        render();
        toast("Sıralama", sortMode === "priceAsc" ? "Fiyat artan." : "Fiyat azalan.");
      });

      $("#quickSortNew").addEventListener("click", () => {
        sortMode = "new";
        $("#sortSelect").value = "new";
        render();
        toast("Sıralama", "Yeni ürünler öne alındı.");
      });

      document.addEventListener("click", (e) => {
        const addBtn = e.target.closest("[data-add]");
        if(addBtn){
          cartCount += 1;
          setCartCount(cartCount);
          $("#cartCount").textContent = cartCount;
          const id = addBtn.getAttribute("data-add");
          const p = PRODUCTS.find(x => x.id === id);
          toast("Sepete eklendi", `${p ? p.name : "Ürün"} eklendi.`);
        }

        const viewBtn = e.target.closest("[data-view]");
        if(viewBtn){
          e.preventDefault();
          // product.html sonraki dosyada gelecek
          toast("Ürün Detay", "product.html sonraki dosyada gelecek.");
        }
      });

      $("#cartBtn").addEventListener("click", () => {
        toast("Sepet", "cart.html / checkout.html sonraki dosyada gelecek.");
      });

      const contactMsg = "İletişim: naturprime0@gmail.com • Instagram: @naturprime";
      $("#navContact").addEventListener("click", (e) => { e.preventDefault(); toast("İletişim", contactMsg); });
      $("#footContact").addEventListener("click", (e) => { e.preventDefault(); toast("İletişim", contactMsg); });

      $("#toastClose").addEventListener("click", () => $("#toast").classList.remove("show"));
    }

    // ===== Start =====
    $("#cartCount").textContent = cartCount;
    setActiveChip("all");
    wire();
    render();
  </script>
