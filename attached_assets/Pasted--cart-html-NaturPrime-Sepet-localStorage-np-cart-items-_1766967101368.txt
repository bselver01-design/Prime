  <script>
    'use strict';

    // ===== Storage =====
    const LS_CART_ITEMS = "np_cart_items_v1";

    function getCart(){
      try{
        const raw = localStorage.getItem(LS_CART_ITEMS);
        return raw ? JSON.parse(raw) : [];
      }catch{ return []; }
    }
    function setCart(items){
      localStorage.setItem(LS_CART_ITEMS, JSON.stringify(items));
    }

    // ===== Helpers =====
    const $ = (s) => document.querySelector(s);

    function moneyTRY(n){
      return new Intl.NumberFormat('tr-TR', { style:'currency', currency:'TRY', maximumFractionDigits:0 }).format(n || 0);
    }

    function calc(){
      const items = getCart();
      const count = items.reduce((a,it)=> a + (it.qty||0), 0);
      const subtotal = items.reduce((a,it)=> a + (Number(it.price)||0) * (it.qty||0), 0);

      // Demo: kargo 0, indirim 0 (istersen sonra kurgu ekleriz)
      const discount = 0;
      const shipping = 0;
      const total = Math.max(0, subtotal - discount + shipping);

      $("#cartCount").textContent = count;
      $("#subtotal").textContent = moneyTRY(subtotal);
      $("#discount").textContent = moneyTRY(discount);
      $("#shipping").textContent = moneyTRY(shipping);
      $("#total").textContent = moneyTRY(total);

      return { items, count, subtotal, total };
    }

    function render(){
      const { items } = calc();
      const wrap = $("#items");
      wrap.innerHTML = "";

      if(items.length === 0){
        wrap.innerHTML = `
          <div class="empty">
            Sepet boş.<br>
            Kataloğa gidip ürün ekleyebilirsin.
          </div>
        `;
        return;
      }

      items.forEach(it => {
        const el = document.createElement("div");
        el.className = "item";
        el.innerHTML = `
          <div class="thumb" aria-hidden="true"></div>

          <div>
            <p class="iname">${it.title || "ÜRÜN"}</p>
            <p class="iprice">${moneyTRY(it.price)}</p>
          </div>

          <div class="controls">
            <div class="qty">
              <button type="button" data-dec="${it.id}">−</button>
              <span>${it.qty || 1}</span>
              <button type="button" data-inc="${it.id}">+</button>
            </div>
            <button class="remove" type="button" data-remove="${it.id}">Sil</button>
          </div>
        `;
        wrap.appendChild(el);
      });
    }

    function changeQty(id, delta){
      const items = getCart();
      const idx = items.findIndex(x => x.id === id);
      if(idx < 0) return;

      const next = (items[idx].qty || 1) + delta;
      if(next <= 0){
        items.splice(idx, 1);
      }else{
        items[idx].qty = Math.min(99, next);
      }
      setCart(items);
      render();
    }

    function removeItem(id){
      const items = getCart().filter(x => x.id !== id);
      setCart(items);
      render();
    }

    // Events
    document.addEventListener("click", (e) => {
      const inc = e.target.closest("[data-inc]");
      if(inc){ changeQty(inc.getAttribute("data-inc"), +1); return; }

      const dec = e.target.closest("[data-dec]");
      if(dec){ changeQty(dec.getAttribute("data-dec"), -1); return; }

      const rem = e.target.closest("[data-remove]");
      if(rem){ removeItem(rem.getAttribute("data-remove")); return; }
    });

    $("#clearCart").addEventListener("click", () => {
      localStorage.removeItem(LS_CART_ITEMS);
      render();
    });
